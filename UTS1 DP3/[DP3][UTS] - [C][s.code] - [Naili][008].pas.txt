begin

write('masukkan bilangan bulat :'); readln(bil);



for i:=bil downto 1 do
begin
    
	for j:=i to bil-1 do 
       
		write(' ');

	writeln;
    
	end;

for i:=2 to bil do 

begin
    
	for j:=i to bil-1 do 
        
		write(' ');
    
	for j:=1 to i do

	writeln;
    
	end;

end.
